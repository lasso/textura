  174.     private function _insertRow($where, array $row, array $colspan = array(), $rowspan = 0, $rowspanPos = 0) {
  175.  
  176.         // @TODO simplify, own Exception
  177.         $colspans = count($colspan);
  178.         if ($colspans > 0 && count($row) != $colspans) {
  179.             throw new Exception(sprintf('Number of columns: %1$s does not match colspans: %2$s', count($row), $colspans));
  180.         }
  181.  
  182.         $rowElm     = new TableElements\Tr();
  183.         $counter = 0;
  184.         foreach ($row as $key => $value) {
  185.  
  186.             if (count($colspan) == 0 && isset($this->lastRowspan[$where][$counter]) && $this->lastRowspan[$where][$counter] == 0 && count($row) != $this->noOfColums) {
  187.                 throw new Exception(sprintf('Number of columns: %1$s does not match initial value: %2$s', count($row), $this->noOfColums));
  188.             }
  189.  
  190.             if (isset($this->lastRowspan[$where][$counter]) && $counter == $rowspanPos && $this->lastRowspan[$where][$counter] > 0) {
  191.  
  192.                 if ($this->lastRowspan[$where][$counter] > 0 && $rowspan != 0) {
  193.                     throw new Exception(sprintf('Rowspan already open!'));
  194.                 }
  195.  
  196.                 $openSpans = 0;
  197.                 foreach ($this->lastRowspan[$where] as $span) {
  198.                     if ($span != 0) {
  199.                         $openSpans++;
  200.                     }
  201.                 }
  202.  
  203.                 $diff = ($this->noOfColums - $openSpans);
  204.                 if (count($row) != $diff) {
  205.                     throw new Exception(sprintf('Cannot insert %1$s columns while rowspan is opened, insert exactly %2$s', count($row), $diff));
  206.                 }
  207.  
  208.                 if (($this->lastRowspanCounter[$where][$counter] - 2) == 0) {
  209.                     $this->lastRowspan[$where][$counter]        = 0;
  210.                     $this->lastRowspanCounter[$where][$counter] = 0;
  211.                 }
  212.  
  213.                 $this->lastRowspanCounter[$where][$counter] = ($this->lastRowspanCounter[$where][$counter] - 1);
  214.             }
  215.  
  216.             $td = new TableElements\Td($value);
  217.  
  218.             if (isset($colspan[$counter]) && $colspan[$counter] != 1) {
  219.                 $td->setColspan($colspan[$counter]);
  220.             }
  221.  
  222.             if (!preg_match("/[0-9]*/is", $key)) {
  223.                 $td->setId($key);
  224.             }
  225.  
  226.             if ($rowspan != 0 && $counter == $rowspanPos) {
  227.                 $td->setRowspan($rowspan);
  228.                 $this->lastRowspan[$where][$counter]        = $rowspan;
  229.                 $this->lastRowspanCounter[$where][$counter] = $rowspan;
  230.                 $rowspan = 0;
  231.             }
  232.  
  233.             $rowElm->insertChild($td);
  234.             $counter++;
  235.         }
  236.         $this->tbody[] = $rowElm;
  237.  
  238.         return $this;
  239.     }
